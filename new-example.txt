

example

; sum from 1 to 100

start:
	addi	r2, r0, r0		; r2 is sum
	addi	r3, r0, 1		; r3 is counter
	addi	r4, r0, 100
loop:
	add		r2, r2, r3
	addi	r3, r3, 1
	slti	r1, r4, r3
	beq		r1, r0, loop
	
	
; mul from 1 to 10

start:
	addi	r2, r0, 1		; r2 is result
	addi	r3, r0, 1		; r3 is counter
	addi	r4, r0, 10		; r4 is limit
loop:
	mul		r2, r3
	mflo	r2
	addi	r3, r3, 1
	slti	r1, r4, r3
	beq		r1, r0, loop
